\chapter{Cross-Module Projects}
\label{ch:projects}

\begin{objectives}
  \item Apply skills from multiple modules in real-world projects
  \item Work independently on extended assignments
  \item Create professional-quality documents, spreadsheets, and presentations
  \item Present your work to the class
\end{objectives}

\bytetip[fun]{These projects combine EVERYTHING you've learned. Think of them as your ``final boss'' challenges! Complete them and you'll be a true computer whiz.}

\section{Project 1: My Digital Portfolio}
\label{sec:proj1}

\begin{labstep}[Project Brief]
Create a \textbf{digital portfolio} showcasing your best work from this course.

\textbf{Deliverables:}
\begin{enumerate}[label=\textcolor{ModuleBlue}{\textbf{\arabic*.}}, itemsep=3pt]
  \item \textbf{Word Document:} A 2-page ``About Me'' profile with formatted text, a table of skills, an inserted photo, headers/footers, and page numbers. Save as PDF.
  \item \textbf{Excel Spreadsheet:} A ``Skills Tracker'' with your self-rated scores (1--10) for 8 skills. Include SUM, AVERAGE formulas, and a bar chart showing your strengths.
  \item \textbf{PowerPoint Presentation:} A 6-slide ``My Journey in Computer Science'' presentation with transitions, animations, and at least one embedded chart or SmartArt.
  \item \textbf{Folder Structure:} Organise all files in a folder called ``MyPortfolio'' with sub-folders: Documents, Spreadsheets, Presentations.
\end{enumerate}

\textbf{Time:} 2--3 class periods\\
\textbf{Assessment:} Presentation quality, proper formatting, organisation, creativity.
\end{labstep}

\section{Project 2: School Event Planner}
\label{sec:proj2}

\begin{labstep}[Project Brief]
Plan a school event (Sports Day, Science Fair, or Cultural Festival) using MS Office.

\textbf{Deliverables:}
\begin{enumerate}[label=\textcolor{ModuleBlue}{\textbf{\arabic*.}}, itemsep=3pt]
  \item \textbf{Word:} An invitation letter to parents with WordArt title, formatted paragraphs, and event details in a table.
  \item \textbf{Excel:} A budget spreadsheet with items, estimated costs, actual costs, and formulas for totals and difference. Include a pie chart of spending categories.
  \item \textbf{PowerPoint:} A 10-slide presentation to pitch the event to your principal. Include timeline (SmartArt), budget summary (chart), and team roles.
  \item \textbf{Email:} Draft a formal email to a teacher inviting them to be a judge, with the invitation letter attached.
\end{enumerate}

\textbf{Time:} 3--4 class periods\\
\textbf{Assessment:} Completeness, professionalism, effective use of all three Office apps.
\end{labstep}

\section{Project 3: Technology Research Report}
\label{sec:proj3}

\begin{labstep}[Project Brief]
Research an emerging technology and create a comprehensive report.

\textbf{Choose one topic:}
\begin{itemize}[leftmargin=*, label={\textcolor{ModuleBlue}{\faAngleRight}}, itemsep=2pt]
  \item Artificial Intelligence (AI)
  \item Virtual Reality (VR)
  \item 3D Printing
  \item Self-Driving Cars
  \item Space Technology
\end{itemize}

\textbf{Deliverables:}
\begin{enumerate}[label=\textcolor{ModuleBlue}{\textbf{\arabic*.}}, itemsep=3pt]
  \item \textbf{Word:} A 3-page research report with: title page, introduction, 3 main sections, conclusion, and a bibliography of at least 3 sources. Use proper formatting, headings, and at least one image.
  \item \textbf{Excel:} A data table with at least 10 facts/statistics about your topic. Create a chart from the data.
  \item \textbf{PowerPoint:} A 8-slide presentation summarising your research for the class.
\end{enumerate}

\textbf{Time:} 3--4 class periods\\
\textbf{Assessment:} Research quality, source reliability (CRAAP test), visual design, presentation skills.
\end{labstep}

\bytetip[tip]{For all projects: plan before you start, save frequently (\key{Ctrl}+\key{S}!), and proofread everything. A little extra effort on presentation makes a \textbf{huge} difference in how your work is received.}

\begin{funfact}
Many technology companies like Google, Apple, and Microsoft still use presentations, spreadsheets, and documents as their primary tools for business communication. The skills you're learning right now are the \emph{exact same skills} used in offices around the world!
\end{funfact}
