\chapter{Part III --- Hands-On Practicals}
\label{ch:lab3}

\bytetip[fun]{These practicals will sharpen your hardware knowledge, file skills, and make you a keyboard shortcut ninja! Let's go!}

% === PRACTICAL 1 ===
\begin{labactivity}{1}{Identifying Internal Hardware}{10 min}
\youwillneed{\faDesktop\ Computer or diagram \quad \faClock\ 10 min \quad \faFile\ Worksheet}

\textbf{Objective:} Identify CPU, RAM, and storage components.\\[4pt]
\textbf{Steps:}
\begin{enumerate}[label=\protect\stepcircle{\arabic*}, leftmargin=2cm, itemsep=4pt]
  \item Look at the diagram your teacher provides showing the inside of a computer.
  \item Label the CPU, RAM sticks, hard drive/SSD, motherboard, and power supply.
  \item For each component, write one sentence about what it does.
  \item Compare RAM and ROM --- fill in a comparison table on your worksheet.
  \item Open Settings $\rightarrow$ About and check your computer's CPU and RAM specs.
\end{enumerate}
\textbf{Expected Outcome:} Students correctly identify and describe 5 internal components.
\welldone
\end{labactivity}

% === PRACTICAL 2 ===
\begin{labactivity}{2}{Exploring File Extensions}{10 min}
\youwillneed{\faDesktop\ Computer \quad \faClock\ 10 min \quad \faFolder\ File Explorer}

\textbf{Objective:} Identify file types by their extensions.\\[4pt]
\textbf{Steps:}
\begin{enumerate}[label=\protect\stepcircle{\arabic*}, leftmargin=2cm, itemsep=4pt]
  \item Open File Explorer and navigate to Documents.
  \item In the View menu, enable ``File name extensions'' so you can see them.
  \item Find at least 5 different file types and record their extensions.
  \item For each file, note the extension and what program opens it.
  \item Create a new text file and try changing its extension to .docx --- what happens?
\end{enumerate}
\textbf{Expected Outcome:} Students identify 5+ file extensions and their associated programs.
\welldone
\end{labactivity}

% === PRACTICAL 3 ===
\begin{labactivity}{3}{Zipping and Unzipping Files}{10 min}
\youwillneed{\faDesktop\ Computer \quad \faClock\ 10 min \quad \faFolder\ Files to compress}

\textbf{Objective:} Compress files into a ZIP archive and extract them.\\[4pt]
\textbf{Steps:}
\begin{enumerate}[label=\protect\stepcircle{\arabic*}, leftmargin=2cm, itemsep=4pt]
  \item Create a folder called ``ZipPractice'' in Documents.
  \item Inside it, create 3 text files with different names and some content.
  \item Right-click the folder $\rightarrow$ ``Compress to ZIP file.''
  \item Note the size of the original folder and the ZIP file. Is the ZIP smaller?
  \item Right-click the ZIP file $\rightarrow$ ``Extract All'' to a new location.
  \item Verify all 3 files are intact in the extracted folder.
\end{enumerate}
\textbf{Expected Outcome:} Students compress and extract files successfully.
\welldone
\end{labactivity}

% === PRACTICAL 4 ===
\begin{labactivity}{4}{Keyboard Typing Practice}{15 min}
\youwillneed{\faDesktop\ Computer \quad \faClock\ 15 min \quad \faKeyboard\ Keyboard}

\textbf{Objective:} Practise typing with correct finger placement.\\[4pt]
\textbf{Steps:}
\begin{enumerate}[label=\protect\stepcircle{\arabic*}, leftmargin=2cm, itemsep=4pt]
  \item Open Notepad.
  \item Place your fingers on the \textbf{home row}: left hand on A-S-D-F, right on J-K-L-;
  \item Type the following without looking at the keyboard: ``the quick brown fox jumps over the lazy dog''
  \item Now type it again, timing yourself. Record your time.
  \item Type it three more times, trying to get faster each time.
  \item Write down your fastest time.
\end{enumerate}
\textbf{Expected Outcome:} Students improve their typing speed over 5 attempts.
\welldone
\end{labactivity}

% === PRACTICAL 5 ===
\begin{labactivity}{5}{Keyboard Shortcuts Practice}{15 min}
\youwillneed{\faDesktop\ Computer \quad \faClock\ 15 min \quad \faKeyboard\ Keyboard}

\textbf{Objective:} Use 10 different keyboard shortcuts in a real task.\\[4pt]
\textbf{Steps:}
\begin{enumerate}[label=\protect\stepcircle{\arabic*}, leftmargin=2cm, itemsep=4pt]
  \item Open \textbf{Notepad} (\key{Win} $\rightarrow$ type ``Notepad'' $\rightarrow$ \key{Enter}).
  \item Type three sentences about your favourite hobby.
  \item Use \key{Ctrl}+\key{A} to select all text.
  \item Use \key{Ctrl}+\key{C} to copy, then \key{Ctrl}+\key{V} to paste it below.
  \item Use \key{Ctrl}+\key{Z} to undo the paste.\item Use \key{Ctrl}+\key{S} to save as ``ShortcutsPractice.txt.''
  \item Use \key{Ctrl}+\key{F} to find the word ``favourite.''
  \item Use \key{Alt}+\key{Tab} to switch to File Explorer.
  \item Use \key{Win}+\key{D} to show the desktop.
  \item Use \key{Win}+\key{E} to open File Explorer again.
\end{enumerate}
\textbf{Expected Outcome:} Students use 10 shortcuts confidently.
\welldone
\end{labactivity}

% === PRACTICAL 6 ===
\begin{labactivity}{6}{Mouse Skills Challenge}{10 min}
\youwillneed{\faDesktop\ Computer \quad \faClock\ 10 min \quad \faMouse\ Mouse}

\textbf{Objective:} Master all mouse actions in a timed challenge.\\[4pt]
\textbf{Steps:}
\begin{enumerate}[label=\protect\stepcircle{\arabic*}, leftmargin=2cm, itemsep=4pt]
  \item \textbf{Single click:} Select 5 different desktop icons (one at a time).
  \item \textbf{Double-click:} Open Recycle Bin, then close it.
  \item \textbf{Right-click:} Right-click the desktop, explore the menu, press Esc.
  \item \textbf{Drag and drop:} Rearrange 3 desktop icons to new positions.
  \item \textbf{Scroll:} Open a long document or web page and scroll to the bottom.
  \item Time yourself doing all 5 tasks. Record your total time.
\end{enumerate}
\textbf{Expected Outcome:} Students complete all mouse actions within 3 minutes.
\welldone
\end{labactivity}
