\section{Creating a Presentation}
\label{sec:create-ppt}
\index{MS PowerPoint!creating}

\begin{labstep}[Creating Your First Presentation]
\begin{enumerate}[label=\protect\stepcircle{\arabic*}, leftmargin=2cm, itemsep=3pt]
  \item Open PowerPoint. Choose a \textbf{theme} or ``Blank Presentation.''
  \item Click ``Click to add title'' and type your title.
  \item Click ``Click to add subtitle'' and type your subtitle.
  \item Add a new slide: Home tab $\rightarrow$ \textbf{New Slide} (or \key{Ctrl}+\key{M}).
  \item Choose a \textbf{slide layout} (Title, Content, Comparison, etc.).
  \item Save as ``MyPresentation.pptx'' (\key{Ctrl}+\key{S}).
\end{enumerate}
\end{labstep}

\begin{theorybox}[Slide Layouts]
PowerPoint offers several built-in slide layouts:
\begin{itemize}[leftmargin=*, label={\textcolor{ModuleBlue}{\faAngleRight}}, itemsep=2pt]
  \item \textbf{Title Slide} --- For the first slide (title + subtitle)
  \item \textbf{Title and Content} --- Title + bullet points or images
  \item \textbf{Two Content} --- Side-by-side content comparison
  \item \textbf{Blank} --- Empty slide for custom designs
  \item \textbf{Section Header} --- To divide your presentation into parts
\end{itemize}
\end{theorybox}

\bytetip[fun]{The average human attention span is about \textbf{8 seconds}. That's why good presentations use visuals, short text, and engaging animations. Nobody wants to read a wall of text on a slide!}

\begin{funfact}
An estimated \textbf{35 million PowerPoint presentations} are created every single day worldwide. That's over 400 presentations every second! Microsoft estimates that about 500 million people use PowerPoint regularly.
\end{funfact}
