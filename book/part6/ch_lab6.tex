\chapter{Part VI --- Hands-On Practicals}
\label{ch:lab6}

\bytetip[fun]{It's showtime! These practicals will turn you into a presentation wizard. Lights, slides, action!}

\begin{labactivity}{1}{Creating Your First Presentation}{15 min}
\youwillneed{\faDesktop\ Computer \quad \faClock\ 15 min \quad \faFilePowerpoint\ MS PowerPoint}
\textbf{Objective:} Create a basic 5-slide presentation.\\[4pt]
\textbf{Steps:}
\begin{enumerate}[label=\protect\stepcircle{\arabic*}, leftmargin=2cm, itemsep=4pt]
  \item Open PowerPoint and choose a theme.
  \item Slide 1: Add a title ``About Me'' and your name as subtitle.
  \item Add 4 more slides (Home $\rightarrow$ New Slide): My Family, My Hobbies, My School, Thank You.
  \item Add bullet points on each content slide.
  \item Save as ``AboutMe.pptx.''
  \item Run the slideshow (\key{F5}) and click through all slides.
\end{enumerate}
\textbf{Expected Outcome:} A 5-slide presentation with consistent theme.
\welldone
\end{labactivity}

\begin{labactivity}{2}{Inserting Images and Shapes}{15 min}
\youwillneed{\faDesktop\ Computer \quad \faClock\ 15 min \quad \faFilePowerpoint\ MS PowerPoint}
\textbf{Objective:} Enhance slides with visual elements.\\[4pt]
\textbf{Steps:}
\begin{enumerate}[label=\protect\stepcircle{\arabic*}, leftmargin=2cm, itemsep=4pt]
  \item Open ``AboutMe.pptx.''
  \item On ``My Hobbies'' slide, insert a picture (Insert $\rightarrow$ Pictures).
  \item Resize the image and position it next to your text.
  \item Insert a shape (star, arrow, or callout) and add text inside.
  \item On another slide, insert SmartArt (Insert $\rightarrow$ SmartArt) to show a process.
  \item Save.
\end{enumerate}
\textbf{Expected Outcome:} Slides with images, shapes, and SmartArt.
\welldone
\end{labactivity}

\begin{labactivity}{3}{Applying Transitions}{10 min}
\youwillneed{\faDesktop\ Computer \quad \faClock\ 10 min \quad \faFilePowerpoint\ MS PowerPoint}
\textbf{Objective:} Add transitions between slides.\\[4pt]
\textbf{Steps:}
\begin{enumerate}[label=\protect\stepcircle{\arabic*}, leftmargin=2cm, itemsep=4pt]
  \item Open ``AboutMe.pptx.''
  \item Click on Slide 1. Go to Transitions tab. Choose ``Fade.''
  \item Click on Slide 2 and choose ``Push.'' Set duration to 1 second.
  \item Apply different transitions to remaining slides.
  \item Click ``Apply To All'' to test --- then undo and keep individual transitions.
  \item Run slideshow to see the transitions in action.
\end{enumerate}
\textbf{Expected Outcome:} Smooth transitions between all slides.
\welldone
\end{labactivity}

\begin{labactivity}{4}{Adding Animations}{15 min}
\youwillneed{\faDesktop\ Computer \quad \faClock\ 15 min \quad \faFilePowerpoint\ MS PowerPoint}
\textbf{Objective:} Animate text and objects within slides.\\[4pt]
\textbf{Steps:}
\begin{enumerate}[label=\protect\stepcircle{\arabic*}, leftmargin=2cm, itemsep=4pt]
  \item On Slide 1, select the title. Animations tab $\rightarrow$ ``Fly In.''
  \item Select the subtitle. Apply ``Fade'' animation.
  \item On a content slide, select the bullet text. Apply ``Appear'' animation.
  \item Click ``Effect Options'' $\rightarrow$ ``By Paragraph'' so bullets appear one by one.
  \item Select a picture. Apply ``Grow \& Turn'' animation.
  \item Open Animation Pane to review the order and timing.
\end{enumerate}
\textbf{Expected Outcome:} Objects animate in sequence within slides.
\welldone
\end{labactivity}

\begin{labactivity}{5}{Presentation on a Topic}{20 min}
\youwillneed{\faDesktop\ Computer \quad \faClock\ 20 min \quad \faFilePowerpoint\ MS PowerPoint}
\textbf{Objective:} Create a complete, polished presentation on a given topic.\\[4pt]
\textbf{Steps:}
\begin{enumerate}[label=\protect\stepcircle{\arabic*}, leftmargin=2cm, itemsep=4pt]
  \item Choose a topic: ``Saving the Environment'' or ``Space Exploration.''
  \item Create 8 slides: Title, Introduction, 4 Content slides, Summary, Thank You.
  \item Use at least: 2 images, 1 SmartArt, 1 table, consistent transitions.
  \item Apply the 6×6 rule (max 6 bullets, max 6 words each).
  \item Add speaker notes at the bottom of each slide (View $\rightarrow$ Notes).
  \item Save and present to a partner using slideshow mode.
\end{enumerate}
\textbf{Expected Outcome:} A professional 8-slide presentation with all elements.
\welldone
\end{labactivity}

\begin{labactivity}{6}{Slideshow Presentation Skills}{10 min}
\youwillneed{\faDesktop\ Computer \quad \faClock\ 10 min \quad \faFilePowerpoint\ MS PowerPoint}
\textbf{Objective:} Present confidently using slideshow tools.\\[4pt]
\textbf{Steps:}
\begin{enumerate}[label=\protect\stepcircle{\arabic*}, leftmargin=2cm, itemsep=4pt]
  \item Open your presentation. Press \key{F5} to start.
  \item Practise advancing slides with clicks and arrow keys.
  \item Press \key{B} to black out the screen. Press any key to return.
  \item Press \key{Shift}+\key{F5} to start from the current slide.
  \item Use the pointer tool (right-click $\rightarrow$ Pointer Options $\rightarrow$ Laser Pointer).
  \item Present to a partner --- speak clearly! Limit slides to 30 seconds each.
\end{enumerate}
\textbf{Expected Outcome:} Students present confidently using slideshow controls.
\welldone
\end{labactivity}
