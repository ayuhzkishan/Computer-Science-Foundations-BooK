\section{Inserting Pictures and Shapes}
\label{sec:insert-pictures}
\index{MS Word!pictures}\index{MS Word!shapes}

\begin{labstep}[Inserting a Picture]
\begin{enumerate}[label=\protect\stepcircle{\arabic*}, leftmargin=2cm, itemsep=3pt]
  \item Click where you want the picture to appear.
  \item Go to \textbf{Insert} tab $\rightarrow$ \textbf{Pictures}.
  \item Choose ``This Device'' to insert from your computer, or ``Online Pictures'' for web images.
  \item Select the image and click \textbf{Insert}.
  \item Resize by dragging the corner handles. Use the \textbf{Layout Options} button to choose how text wraps around the image.
\end{enumerate}
\end{labstep}

\section{Headers, Footers, and Page Numbers}
\label{sec:headers-footers}
\index{MS Word!headers}\index{MS Word!page numbers}

\begin{theorybox}[Headers and Footers]
\begin{itemize}[leftmargin=*, label={\textcolor{ModuleBlue}{\faAngleRight}}, itemsep=3pt]
  \item A \textbf{Header} appears at the \textbf{top} of every page (e.g., document title, your name)
  \item A \textbf{Footer} appears at the \textbf{bottom} of every page (e.g., page number, date)
  \item To add them: \textbf{Insert} tab $\rightarrow$ \textbf{Header} or \textbf{Footer} or \textbf{Page Number}
\end{itemize}
\end{theorybox}

\bytetip[fun]{You can insert \textbf{WordArt} to create fancy decorated text for titles and headings. Go to Insert $\rightarrow$ WordArt and choose a style. It's like having a graphic designer inside your computer!}

\begin{funfact}
The WordArt feature was introduced in \textbf{Word 97} and became an instant hit. Students in the late '90s used it on literally everything --- to the point where teachers had to \emph{ban} excessive WordArt usage in assignments!
\end{funfact}
