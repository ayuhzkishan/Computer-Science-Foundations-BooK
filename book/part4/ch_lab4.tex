\chapter{Part IV --- Hands-On Practicals}
\label{ch:lab4}

\bytetip[fun]{Time to become a Word wizard! These practicals take you from basic typing all the way to creating beautifully formatted documents.}

\begin{labactivity}{1}{Creating Your First Document}{10 min}
\youwillneed{\faDesktop\ Computer \quad \faClock\ 10 min \quad \faFileWord\ MS Word}
\textbf{Objective:} Open Word, type content, and save a document.\\[4pt]
\textbf{Steps:}
\begin{enumerate}[label=\protect\stepcircle{\arabic*}, leftmargin=2cm, itemsep=4pt]
  \item Open MS Word. Click ``Blank document.''
  \item Type a 5-sentence paragraph about your favourite animal.
  \item Save the document as ``MyFirstDoc.docx'' (\key{Ctrl}+\key{S}).
  \item Close Word. Reopen Word and open your saved file (\key{Ctrl}+\key{O}).
  \item Verify your text is still there.
\end{enumerate}
\textbf{Expected Outcome:} Students create and save their first Word document.
\welldone
\end{labactivity}

\begin{labactivity}{2}{Text Selection and Editing}{10 min}
\youwillneed{\faDesktop\ Computer \quad \faClock\ 10 min \quad \faFileWord\ MS Word}
\textbf{Objective:} Select, copy, cut, paste, and rearrange text.\\[4pt]
\textbf{Steps:}
\begin{enumerate}[label=\protect\stepcircle{\arabic*}, leftmargin=2cm, itemsep=4pt]
  \item Open ``MyFirstDoc.docx'' from Practical 1.
  \item Double-click a word to select it. Press \key{Ctrl}+\key{B} to bold it.
  \item Select the entire first sentence (\key{Ctrl}+click). Copy it (\key{Ctrl}+\key{C}).
  \item Place cursor at the end. Paste (\key{Ctrl}+\key{V}).
  \item Use \key{Ctrl}+\key{Z} to undo. Use \key{Ctrl}+\key{H} to find and replace one word.
\end{enumerate}
\textbf{Expected Outcome:} Students edit text using shortcuts confidently.
\welldone
\end{labactivity}

\begin{labactivity}{3}{Paragraph Formatting}{15 min}
\youwillneed{\faDesktop\ Computer \quad \faClock\ 15 min \quad \faFileWord\ MS Word}
\textbf{Objective:} Apply alignment, spacing, and indentation.\\[4pt]
\textbf{Steps:}
\begin{enumerate}[label=\protect\stepcircle{\arabic*}, leftmargin=2cm, itemsep=4pt]
  \item Create a new document with a title and 3 paragraphs.
  \item Centre-align the title (\key{Ctrl}+\key{E}).
  \item Justify the paragraphs (\key{Ctrl}+\key{J}).
  \item Change line spacing to 1.5 (Home $\rightarrow$ Line Spacing).
  \item Add first-line indent to each paragraph (Layout $\rightarrow$ Indent).
  \item Save as ``FormattedParagraphs.docx.''
\end{enumerate}
\textbf{Expected Outcome:} A professionally formatted 3-paragraph document.
\welldone
\end{labactivity}

\begin{labactivity}{4}{Text Formatting Showcase}{15 min}
\youwillneed{\faDesktop\ Computer \quad \faClock\ 15 min \quad \faFileWord\ MS Word}
\textbf{Objective:} Apply font formatting to create a visually appealing document.\\[4pt]
\textbf{Steps:}
\begin{enumerate}[label=\protect\stepcircle{\arabic*}, leftmargin=2cm, itemsep=4pt]
  \item Type a title: ``My Formatting Showcase.'' Make it \textbf{bold}, size 22, colour blue.
  \item Type a subtitle in \textit{italic}, size 14, colour grey.
  \item Type a paragraph and apply: different font (Arial), size 12, justified.
  \item Highlight one important phrase using the highlight tool (yellow).
  \item Create a bulleted list of 4 items and a numbered list of 4 steps.
  \item Save as ``FormatShowcase.docx.''
\end{enumerate}
\textbf{Expected Outcome:} A document demonstrating 5+ formatting techniques.
\welldone
\end{labactivity}

\begin{labactivity}{5}{Working with Tables}{15 min}
\youwillneed{\faDesktop\ Computer \quad \faClock\ 15 min \quad \faFileWord\ MS Word}
\textbf{Objective:} Insert, format, and populate a table.\\[4pt]
\textbf{Steps:}
\begin{enumerate}[label=\protect\stepcircle{\arabic*}, leftmargin=2cm, itemsep=4pt]
  \item Create a new document. Type a title: ``Class Timetable.''
  \item Insert a 6-column, 7-row table (Insert $\rightarrow$ Table).
  \item Fill in: Column headers = Day, Period 1--5. Rows = Monday--Friday + subjects.
  \item Bold the header row and shade it blue (Table Design tab).
  \item Apply alternating row colours (Table Design $\rightarrow$ Banded Rows).
  \item Save as ``Timetable.docx.''
\end{enumerate}
\textbf{Expected Outcome:} A colour-formatted class timetable.
\welldone
\end{labactivity}

\begin{labactivity}{6}{Inserting Pictures and Shapes}{15 min}
\youwillneed{\faDesktop\ Computer \quad \faClock\ 15 min \quad \faFileWord\ MS Word}
\textbf{Objective:} Add visual elements to a document.\\[4pt]
\textbf{Steps:}
\begin{enumerate}[label=\protect\stepcircle{\arabic*}, leftmargin=2cm, itemsep=4pt]
  \item Create a new document. Add a WordArt title at the top.
  \item Insert a picture from your computer (Insert $\rightarrow$ Pictures).
  \item Resize the picture and set text wrapping to ``Square.''
  \item Insert a shape (Insert $\rightarrow$ Shapes) and add text inside it.
  \item Insert a table below the picture with 3 columns and 4 rows.
  \item Save as ``VisualDoc.docx.''
\end{enumerate}
\textbf{Expected Outcome:} A document with WordArt, picture, shape, and table.
\welldone
\end{labactivity}

\begin{labactivity}{7}{Headers, Footers, and Page Numbers}{10 min}
\youwillneed{\faDesktop\ Computer \quad \faClock\ 10 min \quad \faFileWord\ MS Word}
\textbf{Objective:} Add headers, footers, and automatic page numbers.\\[4pt]
\textbf{Steps:}
\begin{enumerate}[label=\protect\stepcircle{\arabic*}, leftmargin=2cm, itemsep=4pt]
  \item Open any document you created in previous practicals.
  \item Insert a \textbf{header}: Insert $\rightarrow$ Header $\rightarrow$ type your name.
  \item Insert a \textbf{footer}: Insert $\rightarrow$ Footer $\rightarrow$ type ``Confidential.''
  \item Insert page numbers: Insert $\rightarrow$ Page Number $\rightarrow$ Bottom of Page.
  \item Double-click outside the header/footer area to return to the document.
  \item Save the file.
\end{enumerate}
\textbf{Expected Outcome:} Document with header, footer, and page numbers on every page.
\welldone
\end{labactivity}

\begin{labactivity}{8}{Complete Document Project}{20 min}
\youwillneed{\faDesktop\ Computer \quad \faClock\ 20 min \quad \faFileWord\ MS Word}
\textbf{Objective:} Create a multi-page, fully formatted document.\\[4pt]
\textbf{Steps:}
\begin{enumerate}[label=\protect\stepcircle{\arabic*}, leftmargin=2cm, itemsep=4pt]
  \item Create a 2-page document about ``My Favourite Festival.''
  \item Page 1: WordArt title, introductory paragraph (justified, 1.5 spacing), inserted picture.
  \item Page 2: Table with 3 columns (Activity, Time, Description), numbered list of traditions.
  \item Add header with your name, footer with page numbers.
  \item Apply a consistent colour theme throughout.
  \item Save as ``FestivalDoc.docx'' and check Print Preview (\key{Ctrl}+\key{P}).
\end{enumerate}
\textbf{Expected Outcome:} A professional 2-page document using all skills learned.
\welldone
\end{labactivity}
